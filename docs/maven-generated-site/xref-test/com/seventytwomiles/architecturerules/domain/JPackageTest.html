108     /***
109      * <p>terminating <tt>*</tt> should only look in one package past the last
110      * package, not any deeper. so <tt>a.b.*</tt> should match <tt>a.b.c</tt>
111      * but NOT <tt>a.b.c.d</tt></p>
112      */
113     public void testMatchPackage_wildcards_pass() throws Exception {
114 
115 
116         final JPackage wild;
117         wild = new JPackage("com.seventytwomiles.project.dao.*");
118 
119         assertTrue(wild.matches(
120                 new JPackage("com.seventytwomiles.project.dao.hibernate")));
121 
122         assertFalse(wild.matches(new JPackage(
123                 "com.seventytwomiles.project.dao.hibernate.mock")));
124     }

128

145     /***
146      * <p>terminating <tt>..*</tt> should match the last package, and all sub
147      * packages. so <tt>a.b..*</tt> should match <tt>a.b.c</tt> and
148      * <tt>a.b.c.d</tt>, <tt>a.b.c.e</tt>, <tt>a.b.c.f</tt> plus
149      * <tt>a.b.c.d.1</tt>, <tt>a.b.c.d.2</tt>, <tt>a.b.c.d.3</tt></p>
150      */
151     public void testMatchSubPackage_wildcards_pass() throws Exception {
152 
153         final JPackage wild;
154         wild = new JPackage("com.seventytwomiles.project..*");
155 
156         assertTrue(wild.matches(
157                 new JPackage("com.seventytwomiles.project.dao.hibernate")));
158 
159         assertTrue(wild.matches(
160                 new JPackage(
161                         "com.seventytwomiles.project.dao.hibernate.mock")));
162 
163         assertTrue(wild.matches(
164                 new JPackage("com.seventytwomiles.project.dao.hibernate")));
165 
166         assertTrue(wild.matches(
167                 new JPackage(
168                         "com.seventytwomiles.project.dao.hibernate.stub")));
169     }
