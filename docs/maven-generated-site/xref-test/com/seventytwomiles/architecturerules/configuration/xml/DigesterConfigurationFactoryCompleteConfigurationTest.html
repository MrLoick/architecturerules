159     public void testProcessConfiguration() throws Exception {
160 
161         final DigesterConfigurationFactory factory
162                 = new DigesterConfigurationFactory();
163 
164         factory.processConfiguration(completeConfiguration);
165 
166 
167         final List sources = new ArrayList();
168         sources.addAll(factory.getSources());
169 
170         assertEquals(4, sources.size());
171 
172         final SourceDirectory source0 = (SourceDirectory) sources.get(0);
173         assertEquals(
174                 "parent-pom" + File.separator + "target" + File.separator + "classes",
175                 source0.getPath());
176         assertFalse(source0.shouldThrowExceptionWhenNotFound());
177 
178         final SourceDirectory source1 = (SourceDirectory) sources.get(1);
179         assertEquals(
180                 "util" + File.separator + "target" + File.separator + "classes",
181                 source1.getPath());
182         assertTrue(source1.shouldThrowExceptionWhenNotFound());
183 
184         final SourceDirectory source2 = (SourceDirectory) sources.get(2);
185         assertEquals(
186                 "web" + File.separator + "target" + File.separator + "classes",
187                 source2.getPath());
188         assertFalse(source2.shouldThrowExceptionWhenNotFound());
189 
190         final SourceDirectory source3 = (SourceDirectory) sources.get(3);
191         assertEquals(
192                 "core" + File.separator + "target" + File.separator + "classes",
193                 source3.getPath());
194         assertFalse(source3.shouldThrowExceptionWhenNotFound());
195 
196 
197         final List rules = new ArrayList();
198         rules.addAll(factory.getRules());
199 
200         assertEquals(3, rules.size());
201 
202         /***
203          * Validate values of the first Rule
204          */
205         final Rule rule0 = (Rule) rules.get(0);
206 
207         /* id */
208         assertEquals("model", rule0.getId());
209 
210         /* comment */
211         assertEquals("Model should remain completely isolated",
212                 rule0.getComment());
213 
214         /* packages */
215         assertEquals(1, rule0.getPackages().size());
216         assertEquals("com.seventytwomiles.pagerank.core.model",
217                 rule0.getPackages().toArray()[0].toString());
218 
219         /* violations */
220         assertEquals(6, rule0.getViolations().size());
221         assertEquals("com.seventytwomiles.pagerank.core.services",
222                 rule0.getViolations().toArray()[0].toString());
223         assertEquals("com.seventytwomiles.pagerank.core.builder",
224                 rule0.getViolations().toArray()[1].toString());
225         assertEquals("com.seventytwomiles.pagerank.core.dao",
226                 rule0.getViolations().toArray()[2].toString());
227         assertEquals("com.seventytwomiles.pagerank.core.strategy",
228                 rule0.getViolations().toArray()[3].toString());
229         assertEquals("com.seventytwomiles.pagerank.core.dao.hibernate",
230                 rule0.getViolations().toArray()[4].toString());
231         assertEquals("com.seventytwomiles.pagerank.util",
232                 rule0.getViolations().toArray()[5].toString());
233 
234         /***
235          * Nothing special about rule1, it is not tested.
236          *
237          * final Rule rule1 = (Rule) rules.get(1);
238          */
239 
240         /***
241          * Validate values of the second Rule
242          */
243         final Rule rule2 = (Rule) rules.get(2);
244 
245         /* id */
246         assertEquals("dao", rule2.getId());
247 
248         /* comment */
249         assertEquals("The dao interface package should rely on nothing.",
250                 rule2.getComment());
251 
252         /* packages */
253         assertEquals(2, rule2.getPackages().size());
254         assertEquals("com.seventytwomiles.pagerank.core.dao",
255                 rule2.getPackages().toArray()[0].toString());
256         assertEquals("com.seventytwomiles.pagerank.core.dao.hibernate",
257                 rule2.getPackages().toArray()[1].toString());
258 
259         /* violations */
260         assertEquals(3, rule2.getViolations().size());
261         assertEquals("com.seventytwomiles.pagerank.core.services",
262                 rule2.getViolations().toArray()[0].toString());
263         assertEquals("com.seventytwomiles.pagerank.core.builder",
264                 rule2.getViolations().toArray()[1].toString());
265         assertEquals("com.seventytwomiles.pagerank.util",
266                 rule2.getViolations().toArray()[2].toString());
267 
268         /***
269          * Test should perform cyclicalDependency
270          * The default is true, so the configuration should make it false.
271          */
272         assertFalse(factory.doCyclicDependencyTest());
273 
274         /***
275          * Test no-packages
276          * The default is false, so the configuration should make it true.
277          */
278         assertTrue(factory.throwExceptionWhenNoPackages());
