134         assertEquals("com.seventytwomiles.pagerank.core.model",

144                 rule0.getViolations().toArray()[2].toString());

146                 rule0.getViolations().toArray()[3].toString());
147         assertEquals("com.seventytwomiles.pagerank.core.dao.hibernate",
148                 rule0.getViolations().toArray()[4].toString());
149         assertEquals("com.seventytwomiles.pagerank.util",

169                 rule1.getPackages().toArray()[1].toString());

173         assertEquals("com.seventytwomiles.pagerank.core.services",
174                 rule1.getViolations().toArray()[0].toString());
