41  /***
42   * <p>Provides support for any type of rule that can be asserted using the
43   * jdepend library</p>

45   * @author mikenereson
46   * @see RulesServiceImpl

53       * <p>log to debug with.</p>
54       *
55       * @parameter log Log
56       */
57      private static final Log log

61       * <p>The <code>Configuration</code> to test against.</p>

82      /***
83       * <p>Constructor that loads up the configuration and loads all the packages
84       * in the source paths</p>
85       *
86       * @param configuration Configuration
87       * @throws SourceNotFoundException when an required source directory does
88       * not exist and when <tt>exception</tt>=<tt>"true"</tt> in the source
89       * configuration
90       * @throws NoPackagesFoundException when none of the source directories
91       * exist and <tt>no-packages</tt>="<tt>ignore</tt>" in the sources
92       * configuration
93       */
94      AbstractArchitecturalRules(final Configuration configuration)
95              throws SourceNotFoundException, NoPackagesFoundException {
96  
97          log.info("instantiating new AbstractArchitecturalRules");
98  
99          this.configuration = configuration;
100 
101         /* instantiate JDepend */
102         jdepend = new JDepend();
103 
104         /* read source paths from configuration file */
105         final Collection sources = this.configuration.getSources();
106 
107         SourceDirectory source;
108 
109         /* add each source to jdepend */
110         for (Iterator sourceIterator = sources.iterator();
111              sourceIterator.hasNext();) {
112 
113             /* read the next source directory */
114             source = (SourceDirectory) sourceIterator.next();
115             addSourceToJdepend(source);
116         }
117 
118         analyze();
119     }

122     /***
123      * <p>Add a sourceDirectory path to JDepend instance. Throws Exception when
124      * sourceDirectory path can not be found, if a given
125      * <code>SourceDirectory</code> is configured to do so.</p>
126      *
127      * @param sourceDirectory a <code>SourceDirectory</code> to read path from
128      */
129     private void addSourceToJdepend(final SourceDirectory sourceDirectory) {
130 
131         final String sourcePath = sourceDirectory.getPath();
132         final StringBuffer message = new StringBuffer();
133 
134         final boolean throwExceptionWhenNotFound
135                 = sourceDirectory.shouldThrowExceptionWhenNotFound();
136 
137         try {
138 
139             jdepend.addDirectory(sourcePath);
140 
141 
142             if (log.isDebugEnabled()) {
143 
144                 message.append("loaded ");
145                 message.append(throwExceptionWhenNotFound ? "required " : "");
146                 message.append("sourceDirectory ");
147                 message.append(new File("").getAbsolutePath());
148                 message.append("//");
149                 message.append(sourcePath);
150 
151                 log.debug(message.toString());
152             }
153 
154         } catch (final IOException e) {
155 
156             /* sourceDirectory not found */
157             if (log.isWarnEnabled()) {
158 
159                 message.append(throwExceptionWhenNotFound ? "required " : "");
160                 message.append("sourceDirectory ");
161                 message.append(new File("").getAbsolutePath());
162                 message.append("//");
163                 message.append(sourcePath);
164                 message.append(" does not exist");
165 
166                 log.warn(message.toString());
167             }
168 
169             if (sourceDirectory.shouldThrowExceptionWhenNotFound()) {
170 
171                 log.error(sourcePath + " was not found", e);
172 
173                 throw new SourceNotFoundException(
174                         sourcePath + " was not found", e);
175             }
176         }
177     }

180     /***
181      * <p>Analyze with JDepend. Call after JDepend knows about all of the source
182      * paths.</p>
183      */
184     private void analyze() {
185 
186         /***
187          * Ask jdepend to analyze each package in each of the source
188          * directories that were added above.
189          */
190         log.debug("fetching packages");
191         packages = jdepend.analyze();
192 
193         log.debug("checking how many packages were found by JDepend");
194 
195         if (packages.isEmpty()) {
196 
197             log.warn("no packages were found with the given configuration. " +
198                     "check your <sources /> configuration");
199 
200             final boolean isConfiguredToThrowExceptionWhenNoPackagesFound
201                     = configuration.shouldThrowExceptionWhenNoPackages();
202 
203             log.debug("throw exception when no packages? "
204                     + isConfiguredToThrowExceptionWhenNoPackagesFound);
205 
206             if (isConfiguredToThrowExceptionWhenNoPackagesFound) {
207 
208                 log.debug("throwing RuntimeException because no packages " +
209                         "were found");
210 
211                 throw new NoPackagesFoundException("no packages were found " +
212                         "with the given configuration. check your <sources /> " +
213                         "configuration");
214             }
215 
216         } else {
217 
218             log.debug("jdepend found " + packages.size()
219                     + " to analyze for cyclic redundancies");
220         }
221     }

235     /***
236      * <p>Test the given layer (a package, but package is java keyword) against
237      * the given Rules</p>
238      *
239      * @param layer String the package to test
240      * @param rules Collection of rules defining which packages the layer may
241      * not use
242      * @throws DependencyConstraintException when a rule is broken
243      * @throws CyclicRedundancyException when cyclic redundancy is found
244      */
245     void testLayeringValid(final String layer,
246                            final Collection rules)
247             throws DependencyConstraintException, CyclicRedundancyException {
248 
249         final Collection packages = jdepend.analyze();
250 
251         log.debug("checking how many packages were found by JDepend");
252 
253         if (packages.isEmpty()) {
254 
255             log.warn("no packages were found with the given configuration. " +
256                     "check your <sources />");
257 
258             final boolean isConfiguredToThrowExceptionWhenNoPackagesFound
259                     = this.configuration.shouldThrowExceptionWhenNoPackages();
260 
261             log.debug("throw exception when no packages? "
262                     + isConfiguredToThrowExceptionWhenNoPackagesFound);
263 
264             if (isConfiguredToThrowExceptionWhenNoPackagesFound) {
265 
266                 log.debug("throwing CyclicRedundancyException");
267 
268                 final String message = "cyclic redundancy does exist";
269                 throw new CyclicRedundancyException(message);
270             }
271 
272         } else {
273 
274             log.debug("jdepend found " + packages.size()
275                     + " to analyze for dependency architecture");
276         }
277 
278         JavaPackage javaPackage;
279 
280         for (Iterator packageIterator = packages.iterator();
281              packageIterator.hasNext();) {
282 
283             javaPackage = (JavaPackage) packageIterator.next();
284 
285             log.debug("checking dependencies on package "
286                     + javaPackage.getName());
287 
288             final String packageName = javaPackage.getName();
289             testEfferentsValid(layer, rules, javaPackage, packageName);
290         }
291     }

294     /***
295      * <p>Test a given layer (java package) against a Collection of
296      * <code>Rules</code></p>
297      *
298      * @param layer String name of the package to test
299      * @param rules Collection of rules defining which packages the given
300      * package may not depend upon
301      * @param jPackage JavaPackage
302      * @param analyzedPackageName String full name
303      * @throws DependencyConstraintException when a rule is broken
304      */
305     private void testEfferentsValid(final String layer, final Collection rules,
306                                     final JavaPackage jPackage,
307                                     final String analyzedPackageName)
308             throws DependencyConstraintException {
309 
310         final Collection efferents = jPackage.getEfferents();
311 
312         JavaPackage efferentPackage;
313 
314         for (Iterator packageIterator = efferents.iterator();
315              packageIterator.hasNext();) {
316 
317             efferentPackage = (JavaPackage) packageIterator.next();
318 
319             if (rules.contains(efferentPackage.getName())
320                     && analyzedPackageName.equals(layer)) {
321 
322                 final String message = analyzedPackageName
323                         + " is not allowed to depend upon "
324                         + efferentPackage.getName();
325 
326                 log.error(message);
327 
328                 throw new DependencyConstraintException(message);
329             }
330         }
331     }
