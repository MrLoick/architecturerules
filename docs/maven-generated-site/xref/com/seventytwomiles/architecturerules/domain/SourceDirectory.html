175         if (notFound == null)

185             throw new IllegalArgumentException(

196          * String value and boolean values coincide
197          */
198         final boolean shouldThrowException
199                 = getNotFound().equalsIgnoreCase(NOT_FOUND_EXCEPTION);
200 
201         setShouldThrowExceptionWhenNotFound(shouldThrowException);
202     }

241     /***
242      * <p>Replaces inappropriate backslash with the appropriate slash based on
243      * the operating system's requirements</p>
244      *
245      * <p>For example, on a Windows system, <tt>src/main/resources</tt> becomes
246      * <tt>src//main//resource</tt></p>
247      *
248      * <p>TODO: this may be able to be replaced with String.replaceAll, but I
249      * couldn't get the regex just right</p>
250      *
251      * <p>This todo/issue is open at <a href="http://code.google.com/p/architecturerules/issues/detail?id=29">issue
252      * 29</a></p>
253      *
254      * @param path String the path to fix
255      * @return String the fixed path
256      */
257     String replaceBackslashForOS(final String path) {
258 
259         final StringBuffer result = new StringBuffer();
260 
261         final StringCharacterIterator iterator
262                 = new StringCharacterIterator(path);
263 
264         char character = iterator.current();
265 
266         final char goal = File.separator.toCharArray()[0];
267         final char target = (goal == '//' ? '/' : '//');
268 
269         while (character != CharacterIterator.DONE) {
270 
271             result.append(character == target ? goal : character);
272             character = iterator.next();
273         }
274 
275         return result.toString();
276     }
