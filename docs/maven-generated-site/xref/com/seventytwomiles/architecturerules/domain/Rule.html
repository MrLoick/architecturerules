132     public boolean addPackage(final String packageName) {
133         Assert.assertNotNull("null packageName can not be added", packageName);
134         Assert.assertFalse("empty packageName can not be added",
135                 packageName.equals(""));
136 
137         final JPackage jPackage = new JPackage(packageName);
138 
139         if (violations.contains(jPackage))
140             throw new IllegalArchitectureRuleException(
141                     "Could not add " + packageName + " package because there " +
142                             "is already a violation with the same package name " +
143                             "for rule " + id);
144 
145         return packages.add(jPackage);
146     }

213     /***
214      * <p>Add a new violation to this <code>Rule</code>.</p>
215      *
216      * @param violation String a package this this Rule's package may NOT depend
217      * upon
218      * @return Rule this <code>Rule</code> to allow for method chaining.
219      * @throws IllegalArchitectureRuleException a RuntimeException when the
220      * violation could not be added because the violation is one of the packages
221      * being checked.
222      */
223     public Rule addViolation(final JPackage violation) {
224 
225         Assert.assertNotNull("null violation can not be added", violation);
226         Assert.assertFalse("empty violation can not be added",
227                 violation.equals(""));
228 
229         if (packages.contains(violation))
230             throw new IllegalArchitectureRuleException(
231                     "Could not add architecture rule violation that creates rule " +
232                             "that says a package can not use itself. Remove " +
233                             "<violation>" + violation + "</violation> " +
234                             "from rule " + id);
235 
236         final boolean added = violations.add(violation);
237 
238 
239         if (added) {
240             log.debug(format("added violation %s to Rule %s",
241                     violation, id));
242         } else {
243             log.warn(format("failed to add violation %s to Rule %s",
244                     violation, id));
245         }
246 
247         return this;
248     }

281     /***
282      * <p>Describes the properties of this rule in an xml-like format.</p>
283      *
284      * @param outputToConsole boolean <tt>true</tt> to output the description to
285      * the console
286      * @return String of xml that describes this <code>Rule</code>.
287      */
288     private String describe(final boolean outputToConsole) {
289         final StringBuffer builder = new StringBuffer();
290 
291         builder.append("<rule>").append("\r\n");
292 
293         builder.append("\t")
294                 .append("<id>")
295                 .append(id)
296                 .append("</id>")
297                 .append("\r\n");
298 
299         builder.append("\t")
300                 .append("<packages>")
301                 .append(packages)
302                 .append("</packages>")
303                 .append("\r\n");
304 
305         builder.append("\t")
306                 .append("<comment>")
307                 .append(comment)
308                 .append("</comment>")
309                 .append("\r\n");
310 
311         builder.append("\t").append("<violations>").append("\r\n");
312 
313         for (final JPackage violation : violations) {
314             builder.append("\t\t").append("<violation>");
315             builder.append(violation.getPath());
316             builder.append("</violation>");
317             builder.append("\r\n");
318         }
319 
320         builder.append("\t").append("</violations>").append("\r\n");
321         builder.append("</rule>").append("\r\n");
322 
323         if (outputToConsole)
324             System.out.println(builder.toString());
325 
326         return builder.toString();
327     }

330     /***
331      * <p>Creates a String representation of this <code>Rule</code>. Useful for
332      * debugging and logging.</p>
333      *
334      * @return String describes this rule at its current state. Such as
335      *         <samp>['dao' for 'com.company.dao, com.company.dao.hibernate']</samp>

338         final String ruleDescription = "['{0}' for {1}] "

340                 .replaceAll("//{1}", describePackages());
341 
342         return ruleDescription;
343     }

346     /***
347      * <p>Creates a String representation of this <tt>packages</tt>. Useful for
348      * debugging and logging.</p>
349      *
350      * @return String describes this Rule's packages. Such as
351      *         <samp>com.company.dao, com.company.dao.hibernate</samp>
352      */
353     public String describePackages() {
354 
355         final StringBuffer packagesDescription = new StringBuffer();
356 
357         final Object[] packagesArray = packages.toArray();
358         final int totalPackages = packagesArray.length;
359 
360         for (int i = 0; i < totalPackages; i++) {
361 
362             final JPackage jPackage = (JPackage) packagesArray[i];
363             final String packageName = jPackage.getPath();
364 
365             packagesDescription
366                     .append(packageName.trim())
367                     .append(" ");
368 
369             if (i + 1 < totalPackages)
370                 packagesDescription.append(",");
371         }
372 
373         final String description = packagesDescription.toString()
374                 .replaceAll("//[", "")
375                 .replaceAll("//]", "");
376 
377         return description;
378     }
