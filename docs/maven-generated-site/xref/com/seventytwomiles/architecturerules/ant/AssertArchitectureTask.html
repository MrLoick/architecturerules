36  /***
37   * <p>Ant task to assert architecture.</p>
38   *
39   * <p>Usage looks like:</p>
40   *
41   * <pre>
42   * &lt;taskdef name="assertArchitecture"
43   *          classname="com.seventytwomiles.architecturerules.ant.AssertArchitectureTask">
44   *    &lt;classpath refid="class.path"/>
45   * &lt;/taskdef>
46   *
47   * &lt;target name="assert-architecture" depends="compile">
48   *    &lt;assertArchitecture configurationFileName="architecture-rules-pass.xml"/>
49   * &lt;/target>
50   * </pre>
51   *
52   * @author mikenereson
53   */
54  public class AssertArchitectureTask extends Task {
55  
56  
57      /***
58       * <p>The name of the configuration file that is in the classpath that holds
59       * the xml configuration. Recommend <samp>architecture-rules.xml</samp></p>
60       *
61       * @parameter configurationFileName String
62       */
63      private String configurationFileName;
64  
65      /***
66       * <p>Reference the configuration that is built by the
67       * <code>ConfiguraitonFactory</code> that reads the configurationFile. This
68       * configuration may be modified.</p>
69       *
70       * @parameter configuration Configuration
71       */
72      final private Configuration configuration = new Configuration();
73  
74  
75      /***
76       * Setter for property 'configurationFileName'.
77       *
78       * @param configurationFileName Value to set for property
79       * 'configurationFileName'.
80       */
81      public void setConfigurationFileName(final String configurationFileName) {
82          this.configurationFileName = configurationFileName;
83      }
84  
85  
86      /***
87       * @throws BuildException
88       * @see Task#execute()
89       */
90      public void execute() throws BuildException {
91  
92          super.execute();
93  
94          if (null == configurationFileName || "".equals(configurationFileName))
95              throw new IllegalStateException(
96                      "set configurationFileName property");
97  
98          /***
99           * 1. load configuration
100          */
101         final ConfigurationFactory configurationFactory
102                 = new DigesterConfigurationFactory(configurationFileName);
103 
104         configuration.getRules().addAll(configurationFactory.getRules());
105 
106         configuration.getSources().addAll(configurationFactory.getSources());
107 
108         configuration.setDoCyclicDependencyTest(
109                 configurationFactory.doCyclicDependencyTest());
110 
111         configuration.setThrowExceptionWhenNoPackages(
112                 configurationFactory.throwExceptionWhenNoPackages());
113 
114         /***
115          * 2. assert configuration rules
116          */
117         UnmodifiableConfiguration unmodifiableConfiguration
118                 = new UnmodifiableConfiguration(configuration);
119 
120         final RulesService rulesService;
121         rulesService = new RulesServiceImpl(unmodifiableConfiguration);
122         rulesService.performRulesTest();
123 
124         /***
125          * 3. check for cyclic dependency, if requested
126          */
127         if (configuration.shouldDoCyclicDependencyTest()) {
128 
129             unmodifiableConfiguration
130                     = new UnmodifiableConfiguration(configuration);
131 
132             final CyclicRedundancyService redundancyService;
133 
134             redundancyService
135                     = new CyclicRedundancyServiceImpl(
136                     unmodifiableConfiguration);
137 
138             redundancyService.performCyclicRedundancyCheck();
139         }
140     }
