81      protected AbstractArchitectureRulesConfigurationTest() {
82  
83          /* 1. load configuration if a configuration file name was provided */
84  
85          final String configurationFileName = getConfigurationFileName();
86  
87          final ConfigurationFactory configurationFactory;
88  
89          if (configurationFileName != null && configurationFileName.length() > 0) {
90  
91              configurationFactory = new DigesterConfigurationFactory(configurationFileName);
92  
93              configuration.getRules().addAll(configurationFactory.getRules());
94              configuration.getSources().addAll(configurationFactory.getSources());
95          }
96      }
97  
98  
99      boolean doTests() {
100 
101         final RulesService rulesService;
102         rulesService = new RulesServiceImpl(new UnmodifiableConfiguration(configuration));
103 
104         final boolean rulesResults = rulesService.performRulesTest();
105 
106         if (configuration.shouldDoCyclicDependencyTest()) {
107 
108             final CyclicRedundencyService redundencyService = new CyclicRedundencyServiceImpl(new UnmodifiableConfiguration(configuration));
109             redundencyService.performCyclicRedundencyCheck();
110         }
111 
112         return rulesResults;
113     }
