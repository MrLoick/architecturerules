108     /***
109      * <p>Create a new <code>ClassPathResource</code> for ClassLoader usage. A
110      * leading slash will be removed, as the ClassLoader resource access methods
111      * will not accept it.</p>
112      *
113      * @param path the absolute path within the classpath
114      * @param classLoader the class loader to load the resource with, or
115      * <code>null</code> for the thread context class loader
116      * @see ClassLoader#getResourceAsStream(String)
117      */
118     public ClassPathResource(String path, final ClassLoader classLoader) {
119 
120         if (null == path || "".equals(path))
121             throw new IllegalArgumentException("path can not be empty or null");
122 
123         if (path.startsWith("/"))
124             path = path.substring(1);
125 
126         this.path = cleanPath(path);
127         this.classLoader = (classLoader != null ? classLoader : ClassUtils.getDefaultClassLoader());
128     }

228     /***
229      * <p>Determine if the given objects are equal, returning <code>true</code>
230      * if both are <code>null</code> or <code>false</code> if only one is
231      * <code>null</code>. <p>Compares arrays with <code>Arrays.equals</code>,
232      * performing an equality check based on the array elements rather than the
233      * array reference.</p>
234      *
235      * @param object1 first Object to compare
236      * @param object2 second Object to compare
237      * @return whether the given objects are equal
238      * @see java.util.Arrays#equals
239      */
240     private boolean nullSafeEquals(final Object object1, final Object object2) {
241 
242         if (object1 == object2)
243             return true;
244 
245         if (object1 == null || object2 == null)
246             return false;
247 
248         if (object1.equals(object2))
249             return true;
250 
251         if (object1 instanceof Object[] && object2 instanceof Object[])
252             return Arrays.equals((Object[]) object1, (Object[]) object2);
253 
254         if (object1 instanceof boolean[] && object2 instanceof boolean[])
255             return Arrays.equals((boolean[]) object1, (boolean[]) object2);
256 
257         if (object1 instanceof byte[] && object2 instanceof byte[])
258             return Arrays.equals((byte[]) object1, (byte[]) object2);
259 
260         if (object1 instanceof char[] && object2 instanceof char[])
261             return Arrays.equals((char[]) object1, (char[]) object2);
262 
263         if (object1 instanceof double[] && object2 instanceof double[])
264             return Arrays.equals((double[]) object1, (double[]) object2);
265 
266         if (object1 instanceof float[] && object2 instanceof float[])
267             return Arrays.equals((float[]) object1, (float[]) object2);
268 
269         if (object1 instanceof int[] && object2 instanceof int[])
270             return Arrays.equals((int[]) object1, (int[]) object2);
271 
272         if (object1 instanceof long[] && object2 instanceof long[])
273             return Arrays.equals((long[]) object1, (long[]) object2);
274 
275         if (object1 instanceof short[] && object2 instanceof short[])
276             return Arrays.equals((short[]) object1, (short[]) object2);
277 
278         return false;
279     }

284      * to whether an InputStream can be opened. This will cover both directories
285      * and content resources.</p>

289     public boolean exists() {
290 
291         // Try file existence: can we find the file in the file system?
292         try {
293 
294             return getFile().exists();
295 
296         } catch (final IOException ex) {
297 
298             // Fall back to stream existence: can we open the stream?
299             try {
300 
301                 final InputStream inputStream = getInputStream();
302                 inputStream.close();
303                 return true;
304 
305             } catch (final Throwable throwable) {
306 
307                 return false;
308             }
309         }

334     /***
335      * <p>Normalize the path by suppressing sequences like "path/.." and inner
336      * simple dots. <p>The result is convenient for path comparison. For other
337      * uses, notice that Windows separators ("\") are replaced by simple
338      * slashes.</p>
339      *
340      * @param path the original path
341      * @return the normalized path
342      */
343     private String cleanPath(final String path) {
344 
345         String pathToUse = replace(path, WINDOWS_FOLDER_SEPARATOR, FOLDER_SEPARATOR);
346 
347         // Strip prefix from path to analyze, to not treat it as part of the
348         // first path element. This is necessary to correctly parse paths like
349         // "file:core/../core/io/Resource.class", where the ".." should just
350         // strip the first "core" directory while keeping the "file:" prefix.
351         int prefixIndex = pathToUse.indexOf(":");
352         String prefix = "";
353 
354         if (prefixIndex != -1) {
355 
356             prefix = pathToUse.substring(0, prefixIndex + 1);
357             pathToUse = pathToUse.substring(prefixIndex + 1);
358         }
359 
360         String[] pathArray = delimitedListToStringArray(pathToUse, FOLDER_SEPARATOR);
361 
362         final List pathElements = new LinkedList();
363         int tops = 0;
364 
365         for (int i = pathArray.length - 1; i >= 0; i--) {
366 
367             if (CURRENT_PATH.equals(pathArray[i])) {
368 
369                 // Points to current directory - drop it.
370 
371             } else if (TOP_PATH.equals(pathArray[i])) {
372 
373                 // Registering top path found.
374                 tops++;
375 
376             } else {
377 
378                 if (tops > 0) {
379 
380                     // Merging path element with corresponding to top path.
381                     tops--;
382 
383                 } else {
384 
385                     // Normal path element found.
386                     pathElements.add(0, pathArray[i]);
387                 }
388             }
389         }
390 
391         // Remaining top paths need to be retained.
392         for (int i = 0; i < tops; i++)
393             pathElements.add(0, TOP_PATH);
394 
395         return prefix + collectionToDelimitedString(pathElements, FOLDER_SEPARATOR);
396     }

399     /***
400      * <p>Replace all occurences of a substring within a string with another
401      * string.</p>
402      *
403      * @param inString String to examine
404      * @param oldPattern String to replace
405      * @param newPattern String to insert
406      * @return a String with the replacements
407      */
408     private static String replace(final String inString, final String oldPattern, final String newPattern) {
409 
410         if (inString == null)
411             return null;
412 
413         if (oldPattern == null || newPattern == null)
414             return inString;
415 
416         final StringBuffer stringBuffer = new StringBuffer();
417 
418         // output StringBuffer we'll build up
419         int pos = 0; // our position in the old string
420         int index = inString.indexOf(oldPattern);
421 
422         // the index of an occurrence we've found, or -1
423         final int patLen = oldPattern.length();
424 
425         while (index >= 0) {
426 
427             stringBuffer.append(inString.substring(pos, index));
428             stringBuffer.append(newPattern);
429             pos = index + patLen;
430             index = inString.indexOf(oldPattern, pos);
431         }
432 
433         stringBuffer.append(inString.substring(pos));
434 
435         // remember to append any characters to the right of a match
436         return stringBuffer.toString();
437     }

440     /***
441      * <p>Convenience method to return a Collection as a delimited (e.g. CSV)
442      * String. E.g. useful for <code>toString()</code> implementations.</p>
443      *
444      * @param collection the Collection to display
445      * @param deliminator the delimiter to use (probably a ",")
446      * @param prefix the String to start each element with
447      * @param suffix the String to end each element with
448      * @return String
449      */
450     private static String collectionToDelimitedString(final Collection collection, final String deliminator, final String prefix, final String suffix) {
451 
452         if (collection.isEmpty())
453             return "";
454 
455         final StringBuffer stringBuffer = new StringBuffer();
456         final Iterator iterator = collection.iterator();
457 
458         while (iterator.hasNext()) {
459             stringBuffer.append(prefix).append(iterator.next()).append(suffix);
460 
461             if (iterator.hasNext())
462                 stringBuffer.append(deliminator);
463         }
464 
465         return stringBuffer.toString();
466     }

482     /***
483      * Take a String which is a delimited list and convert it to a String array.
484      * <p>A single delimiter can consists of more than one character: It will
485      * still be considered as single delimiter string, rather than as bunch of
486      * potential delimiter characters - in contrast to <code>tokenizeToStringArray</code>.
487      *
488      * @param string the input String
489      * @param delimiter the delimiter between elements (this is a single
490      * delimiter, rather than a bunch individual delimiter characters)
491      * @return an array of the tokens in the list
492      */
493     private static String[] delimitedListToStringArray(final String string, final String delimiter) {
494 
495         if (string == null)
496             return new String[0];
497 
498         if (delimiter == null)
499             return new String[]{string};
500 
501         final List result = new ArrayList();
502 
503         if ("".equals(delimiter)) {
504 
505             for (int i = 0; i < string.length(); i++)
506                 result.add(string.substring(i, i + 1));
507 
508         } else {
509 
510             int position = 0;
511             int deletePosition;
512 
513             while ((deletePosition = string.indexOf(delimiter, position)) != -1) {
514 
515                 result.add(string.substring(position, deletePosition));
516                 position = deletePosition + delimiter.length();
517             }
518 
519             // Add rest of String, but not in case of empty input.
520             if (string.length() > 0 && position <= string.length())
521                 result.add(string.substring(position));
522         }
523 
524         return toStringArray(result);
525     }
