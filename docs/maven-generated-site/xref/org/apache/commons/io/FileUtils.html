55      /***
56       * <p> Reads the contents of a file into a String. There is no
57       * readFileToString method without encoding parameter because the default
58       * encoding can differ between platforms and therefore results in
59       * inconsistent results. </p>
60       *
61       * @param file the file to read
62       * @param encoding the encoding to use, null means platform default
63       * @return The file contents or null if read failed.
64       * @throws IOException in case of an I/O error
65       * @throws UnsupportedEncodingException if the encoding is not supported by
66       * the VM
67       */
68      public static String readFileToString(final File file, final String encoding) throws IOException {
69  
70          final InputStream inputStream = new FileInputStream(file);
71  
72          try {
73  
74              return toString(inputStream, encoding);
75  
76          } finally {
77  
78              closeQuietly(inputStream);
79          }
80      }

83      /***
84       * <p>Get the contents of an <code>InputStream</code> as a String using the
85       * specified character encoding. <p> Character encoding names can be found
86       * at <a href="http://www.iana.org/assignments/character-sets">IANA</a>. <p>
87       * This method buffers the input internally, so there is no need to use a
88       * <code>BufferedInputStream</code>.</p>
89       *
90       * @param input the <code>InputStream</code> to read from
91       * @param encoding the encoding to use, null means platform default
92       * @return the requested String
93       * @throws NullPointerException if the input is null
94       * @throws IOException if an I/O error occurs
95       */
96      public static String toString(final InputStream input, final String encoding) throws IOException {
97  
98          final StringWriter stringWriter = new StringWriter();
99  
100         copy(input, stringWriter, encoding);
101 
102         return stringWriter.toString();
103     }

107      * <p>Copy bytes from an <code>InputStream</code> to chars on a
108      * <code>Writer</code> using the default character encoding of the platform.

126     /***
127      * <p>Copy bytes from an <code>InputStream</code> to chars on a
128      * <code>Writer</code> using the specified character encoding. <p> This
129      * method buffers the inputStream internally, so there is no need to use a
130      * <code>BufferedInputStream</code>. <p> Character encoding names can be
131      * found at <a href="http://www.iana.org/assignments/character-sets">IANA</a>.
132      * <p> This method uses {@link InputStreamReader}.</p>
133      *
134      * @param inputStream the <code>InputStream</code> to read from
135      * @param outputStream the <code>Writer</code> to write to
136      * @param encoding the encoding to use, null means platform default
137      * @throws NullPointerException if the inputStream or outputStream is null
138      * @throws IOException if an I/O error occurs
139      * @since Commons IO 1.1
140      */
141     public static void copy(final InputStream inputStream, final Writer outputStream, final String encoding) throws IOException {
142 
143         if (encoding == null) {
144 
145             copy(inputStream, outputStream);
146 
147         } else {
148 
149             final InputStreamReader inputStreamReader = new InputStreamReader(inputStream, encoding);
150             copy(inputStreamReader, outputStream);
151         }
152     }

155     /***
156      * <p>Copy chars from a <code>Reader</code> to a <code>Writer</code>. <p>
157      * This method buffers the input internally, so there is no need to use a
158      * <code>BufferedReader</code>.</p>
159      *
160      * @param input the <code>Reader</code> to read from
161      * @param output the <code>Writer</code> to write to
162      * @return the number of characters copied
163      * @throws NullPointerException if the input or output is null
164      * @throws IOException if an I/O error occurs
165      * @since Commons IO 1.1
166      */
167     public static int copy(Reader input, Writer output) throws IOException {
168 
169         final char[] buffer = new char[DEFAULT_BUFFER_SIZE];
170 
171         int count = 0;
172         int n;
173 
174         while (-1 != (n = input.read(buffer))) {
175 
176             output.write(buffer, 0, n);
177             count += n;
178         }
179 
180         return count;
181     }

184     /***
185      * <code>Copy chars from a <code>Reader</code> to bytes on an
186      * <code>OutputStream</code> using the default character encoding of the
187      * platform, and calling flush. <p> This method buffers the input
188      * internally, so there is no need to use a <code>BufferedReader</code>. <p>
189      * Due to the implementation of OutputStreamWriter, this method performs a
190      * flush. <p> This method uses {@link OutputStreamWriter}.</code>
191      *
192      * @param input the <code>Reader</code> to read from
193      * @param output the <code>OutputStream</code> to write to
194      * @throws NullPointerException if the input or output is null
195      * @throws IOException if an I/O error occurs
196      * @since Commons IO 1.1
197      */
198     public static void copy(final Reader input, final OutputStream output) throws IOException {
199 
200         final OutputStreamWriter out = new OutputStreamWriter(output);
201 
202         copy(input, out);
203         out.flush();
204     }

207     /***
208      * <code>Unconditionally close an <code>InputStream</code>. <p> Equivalent
209      * to {@link InputStream#close()}, except any exceptions will be ignored.
210      * This is typically used in finally blocks.</code>
211      *
212      * @param input the InputStream to close, may be null or already closed
213      */
214     public static void closeQuietly(final InputStream input) {
215 
216         try {
217 
218             if (input != null)
219                 input.close();
220 
221         } catch (IOException ioe) {
222 
223             // ignore
224         }
225     }
