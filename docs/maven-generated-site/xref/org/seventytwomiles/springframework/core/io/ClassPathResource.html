104     /***
105      * <p>Create a new <code>ClassPathResource</code> for ClassLoader usage. A
106      * leading slash will be removed, as the ClassLoader resource access methods
107      * will not accept it.</p>
108      *
109      * @param path the absolute path within the classpath
110      * @param classLoader the class loader to load the resource with, or
111      * <code>null</code> for the thread context class loader
112      * @see ClassLoader#getResourceAsStream(String)
113      */
114     public ClassPathResource(String path, final ClassLoader classLoader) {
115 
116         if (null == path || "".equals(path))
117             throw new IllegalArgumentException("path can not be empty or null");
118 
119         if (path.startsWith("/"))
120             path = path.substring(1);
121 
122         this.path = cleanPath(path);
123         this.classLoader
124                 = (classLoader != null ? classLoader : ClassUtils.getDefaultClassLoader());
125     }

230     /***
231      * <p>Determine if the given objects are equal, returning <code>true</code>
232      * if both are <code>null</code> or <code>false</code> if only one is
233      * <code>null</code>. <p>Compares arrays with <code>Arrays.equals</code>,
234      * performing an equality check based on the array elements rather than the
235      * array reference.</p>
236      *
237      * @param object1 first Object to compare
238      * @param object2 second Object to compare
239      * @return whether the given objects are equal
240      * @see java.util.Arrays#equals
241      */
242     private boolean nullSafeEquals(final Object object1, final Object object2) {
243 
244         if (object1 == object2)
245             return true;
246 
247         if (object1 == null || object2 == null)
248             return false;
249 
250         if (object1.equals(object2))
251             return true;
252 
253         if (object1 instanceof Object[] && object2 instanceof Object[])
254             return Arrays.equals((Object[]) object1, (Object[]) object2);
255 
256         if (object1 instanceof boolean[] && object2 instanceof boolean[])
257             return Arrays.equals((boolean[]) object1, (boolean[]) object2);
258 
259         if (object1 instanceof byte[] && object2 instanceof byte[])
260             return Arrays.equals((byte[]) object1, (byte[]) object2);
261 
262         if (object1 instanceof char[] && object2 instanceof char[])
263             return Arrays.equals((char[]) object1, (char[]) object2);
264 
265         if (object1 instanceof double[] && object2 instanceof double[])
266             return Arrays.equals((double[]) object1, (double[]) object2);
267 
268         if (object1 instanceof float[] && object2 instanceof float[])
269             return Arrays.equals((float[]) object1, (float[]) object2);
270 
271         if (object1 instanceof int[] && object2 instanceof int[])
272             return Arrays.equals((int[]) object1, (int[]) object2);
273 
274         if (object1 instanceof long[] && object2 instanceof long[])
275             return Arrays.equals((long[]) object1, (long[]) object2);
276 
277         if (object1 instanceof short[] && object2 instanceof short[])
278             return Arrays.equals((short[]) object1, (short[]) object2);
279 
280         return false;
281     }

285      * <p>This implementation checks whether a File can be opened, falling back
286      * to whether an InputStream can be opened. This will cover both directories
287      * and content resources.</p>
288      *
289      * @return boolean
290      */
291     @SuppressWarnings({"BooleanMethodIsAlwaysInverted"})
292     public boolean exists() {
293 
294         // Try file existence: can we find the file in the file system?
295         try {
296 
297             return getFile().exists();
298 
299         } catch (final IOException ex) {
300 
301             // Fall back to stream existence: can we open the stream?
302             try {
303 
304                 final InputStream inputStream = getInputStream();
305                 inputStream.close();
306                 return true;
307 
308             } catch (final Throwable throwable) {
309 
310                 return false;
311             }
312         }
313     }

340     /***
341      * <p>Normalize the path by suppressing sequences like "path/.." and inner
342      * simple dots. <p>The result is convenient for path comparison. For other
343      * uses, notice that Windows separators ("\") are replaced by simple
344      * slashes.</p>
345      *
346      * @param path the original path
347      * @return the normalized path
348      */
349     private String cleanPath(final String path) {
350 
351         String pathToUse = replace(path, WINDOWS_FOLDER_SEPARATOR,
352                 FOLDER_SEPARATOR);
353 
354         // Strip prefix from path to analyze, to not treat it as part of the
355         // first path element. This is necessary to correctly parse paths like
356         // "file:core/../core/io/Resource.class", where the ".." should just
357         // strip the first "core" directory while keeping the "file:" prefix.
358         final int prefixIndex = pathToUse.indexOf(":");
359         String prefix = "";
360 
361         if (prefixIndex != -1) {
362 
363             prefix = pathToUse.substring(0, prefixIndex + 1);
364             pathToUse = pathToUse.substring(prefixIndex + 1);
365         }
366 
367         final String[] pathArray = delimitedListToStringArray(pathToUse,
368                 FOLDER_SEPARATOR);
369 
370         final List pathElements = new LinkedList();
371         int tops = 0;
372 
373         for (int i = pathArray.length - 1; i >= 0; i--) {
374 
375             if (CURRENT_PATH.equals(pathArray[i])) {
376 
377                 // Points to current directory - drop it.
378 
379             } else if (TOP_PATH.equals(pathArray[i])) {
380 
381                 // Registering top path found.
382                 tops++;
383 
384             } else {
385 
386                 if (tops > 0) {
387 
388                     // Merging path element with corresponding to top path.
389                     tops--;
390 
391                 } else {
392 
393                     // Normal path element found.
394                     pathElements.add(0, pathArray[i]);
395                 }
396             }
397         }
398 
399         // Remaining top paths need to be retained.
400         for (int i = 0; i < tops; i++)
401             pathElements.add(0, TOP_PATH);
402 
403         return prefix + collectionToDelimitedString(pathElements,
404                 FOLDER_SEPARATOR);
405     }

408     /***
409      * <p>Replace all occurences of a substring within a string with another
410      * string.</p>
411      *
412      * @param inString String to examine
413      * @param oldPattern String to replace
414      * @param newPattern String to insert
415      * @return a String with the replacements
416      */
417     private static String replace(final String inString,
418                                   final String oldPattern,
419                                   final String newPattern) {
420 
421         if (inString == null)
422             return null;
423 
424         if (oldPattern == null || newPattern == null)
425             return inString;
426 
427         final StringBuffer stringBuffer = new StringBuffer();
428 
429         // output StringBuffer we'll build up
430         int position = 0; // our position in the old string
431         int index = inString.indexOf(oldPattern);
432 
433         // the index of an occurrence we've found, or -1
434         final int patternLength = oldPattern.length();
435 
436         while (index >= 0) {
437 
438             stringBuffer.append(inString.substring(position, index));
439             stringBuffer.append(newPattern);
440             position = index + patternLength;
441             index = inString.indexOf(oldPattern, position);
442         }
443 
444         stringBuffer.append(inString.substring(position));
445 
446         // remember to append any characters to the right of a matches
447         return stringBuffer.toString();
448     }

451     /***
452      * <p>Convenience method to return a Collection as a delimited (e.g. CSV)
453      * String. E.g. useful for <code>toString()</code> implementations.</p>
454      *
455      * @param collection the Collection to display
456      * @param deliminator the delimiter to use (probably a ",")
457      * @param prefix the String to start each element with
458      * @param suffix the String to end each element with
459      * @return String
460      */
461     private static String collectionToDelimitedString(
462             final Collection collection, final String deliminator,
463             final String prefix, final String suffix) {
464 
465         if (collection.isEmpty())
466             return "";
467 
468         final StringBuffer stringBuffer = new StringBuffer();
469         final Iterator iterator = collection.iterator();
470 
471         while (iterator.hasNext()) {
472             stringBuffer.append(prefix).append(iterator.next()).append(suffix);
473 
474             if (iterator.hasNext())
475                 stringBuffer.append(deliminator);
476         }
477 
478         return stringBuffer.toString();
479     }

496     /***
497      * Take a String which is a delimited list and convert it to a String array.
498      * <p>A single delimiter can consists of more than one character: It will
499      * still be considered as single delimiter string, rather than as bunch of
500      * potential delimiter characters - in contrast to <code>tokenizeToStringArray</code>.
501      *
502      * @param string the input String
503      * @param delimiter the delimiter between elements (this is a single
504      * delimiter, rather than a bunch individual delimiter characters)
505      * @return an array of the tokens in the list
506      */
507     private static String[] delimitedListToStringArray(final String string,
508                                                        final String delimiter) {
509 
510         if (string == null)
511             return new String[0];
512 
513         if (delimiter == null)
514             return new String[]{string};
515 
516         final List result = new ArrayList();
517 
518         if ("".equals(delimiter)) {
519 
520             for (int i = 0; i < string.length(); i++)
521                 result.add(string.substring(i, i + 1));
522 
523         } else {
524 
525             int position = 0;
526             int deletePosition;
527 
528             while ((deletePosition = string.indexOf(delimiter,
529                     position)) != -1) {
530 
531                 result.add(string.substring(position, deletePosition));
532                 position = deletePosition + delimiter.length();
533             }
534 
535             // Add rest of String, but not in case of empty input.
536             if (string.length() > 0 && position <= string.length())
537                 result.add(string.substring(position));
538         }
539 
540         return toStringArray(result);
541     }
